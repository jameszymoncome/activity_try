HTML Exercise for Script

Exercise 1: Change the background color of the body elements to "blue".

<body>
  <script>
    document.body.style.backgroundColor = "";
   
  </script>
</body>

Exercise 2: Change the text color of the h1 element to "red"

<body>
  <h1 id="heading">Hello, world!</h1>
  <script>
    document.getElementById("heading").style.color = "";
    
  </script>
</body>

Exercise 3: Hide the img element with the id "myImage".

<body>
  <img id="myImage" src="image.jpg" alt="Image">
  <script>
    document.getElementById("myImage").style.display = "";
   
  </script>
</body>

Exercise 4: Set the border of all div elements to "2px solid green"

<body>
  <div class="box">Box 1</div>
  <div class="box">Box 2</div>
  <script>
    var divList = document.getElementsByTagName("div");
    for (var i = 0; i < divList.length; i++) {
      divList[i].style.border = "";
    }
    
  </script>
</body>

Exercise 5: Change the font family of all p elements to "Arial".

<body>
  <p>Paragraph 1</p>
  <p>Paragraph 2</p>
  <script>
    var pList = document.getElementsByTagName("p");
    for (var i = 0; i < pList.length; i++) {
      pList[i].style.fontFamily = "";
    }
    
  </script>
</body>
